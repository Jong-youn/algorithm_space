## 문제  
ch. 7 - 4  
동전 바꿔주기(DFS)  

## 풀이
```python
import sys
sys.stdin = open("input4.txt", "r")
t = int(input())
k = int(input())
p, n = [], []
for _ in range(k) :
    a, b = map(int, input().split())
    p.append(a)
    n.append(b)
    
cnt = 0

def abo(lv, sum):
    global cnt
    if sum == 0 :
        cnt += 1
        return 
    
    elif sum < 0 or lv == k:
        return 
    
    for i in range(n[lv] + 1) :
        abo(lv + 1, sum - p[lv]*i)
        
    
abo(0, t)
print(cnt)
```
