피자배달거리

## 풀이
```python
import sys
sys.stdin = open("input17.txt", "r")

n, m = map(int, input().split())
town = [[0] * (n+1)]
town += [ [0] + list(map(int, input().split())) for _ in range(n)]
route = dict()
pzas = []
hus = []

for i in range(1, n+1) :            #가정집, 피자집 위치 찾기
    for j in range(1, n+1) :
        if town[i][j] == 2 :
            pzas.append((i,j))
        elif town[i][j] == 1 :
            hus.append((i, j))

res = [0]*m
def part(lv, ls) :
    if lv == m :
        for j in res :
            
    else :
        for i in range(ls, len(pzas)):
            res[lv] = i
            part(lv+1, i+1)

def rt(ox, oy) :
    sum = 0
    for i in range(1, n+1) :
        for j in range(1, n+1) :
            if town[i][j] == 1 :
                df = abs(ox - i) + abs(oy - j)
                sum += df
    route[(ox, oy)] = sum
    
for x, y in pzas :
    rt(x, y)
    
part(0,0)

```
